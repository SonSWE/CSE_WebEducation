CREATE FUNCTION [dbo].[FN_CheckContainList]
(
	@p_parent_array VARCHAR(MAX),
    @p_child_array VARCHAR(MAX)
)
RETURNS INT
AS
BEGIN
	DECLARE @v_count INT = 0;
	DECLARE v_cur_condition CURSOR FOR 
		SELECT [str] AS condition FROM FN_ParserStrToRow(@p_child_array,',');

	DECLARE @item NVARCHAR(1000) = '';
	OPEN v_cur_condition
	FETCH NEXT FROM v_cur_condition INTO @item
	WHILE @@FETCH_STATUS = 0
	BEGIN
		IF CHARINDEX(@item, @p_parent_array) > 0 BEGIN
			SET @v_count = @v_count + 1;
		END;
		FETCH NEXT FROM v_cur_condition INTO @item
	END;
	IF @v_count > 0 
        RETURN 1;
    ELSE
        RETURN 0;
	RETURN @v_count;
END
GO
/****** Object:  UserDefinedFunction [dbo].[FN_ParserStrToRow]    Script Date: 15/03/2023 11:01:52 PM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE FUNCTION [dbo].[FN_ParserStrToRow]
(
    @in_string NVARCHAR(MAX),
    @delimeter NVARCHAR(1)
)
RETURNS @list TABLE(str NVARCHAR(100))
AS
BEGIN
        WHILE LEN(@in_string) > 0
        BEGIN
            INSERT INTO @list(str)
            SELECT left(@in_string, charindex(@delimeter, @in_string+@delimeter) -1) as str
    
            SET @in_string = stuff(@in_string, 1, charindex(@delimeter, @in_string + @delimeter), '')
        end
    RETURN 
END
GO

CREATE PROCEDURE [dbo].[sp_group_search]
(
		@p_user_name NVARCHAR(50),
		@p_key_search NVARCHAR(max),
        @p_startrow INT,
        @p_endrow INT,
        @p_orderby VARCHAR(200),
        @p_total_record DECIMAL(18,0) OUT
)
AS
BEGIN
		DECLARE v_cur_condition CURSOR FOR 
		SELECT [str] AS condition FROM [dbo].[FN_ParserStrToRow](@p_key_search,'|');
		
        DECLARE @v_condition NVARCHAR(max) = '';
		DECLARE @v_sql NVARCHAR(max);
        DECLARE @v_sqlTableParamSearch NVARCHAR(max);
        DECLARE @v_sqlTotal_record NVARCHAR(max);
        DECLARE @v_orderby NVARCHAR(max) = 'ORDER BY ';

        DECLARE @v_group_name NVARCHAR(max) = '';
        DECLARE @v_group_type NVARCHAR(max) = '';

		--DECLARE @v_user_type decimal(1,0);
        
		---- lay thong tin user type
  --      SELECT @v_user_type = a.User_Type FROM CSE_Users A WHERE A.USER_NAME = @p_user_name AND A.DELETED = 0;

		--IF @v_user_type != 0 BEGIN
  --         SET @v_condition = @v_condition + ' AND A.created_by = cd.p_user_name ';
  --      END;

		--SET ORDER BY
		IF LEN(@p_orderby) > 0 
           SET @v_orderby = @v_orderby + @p_orderby;
        ELSE BEGIN
           SET @v_orderby = @v_orderby + ' Group_Id DESC';
        END;

		--GET AND SET INFO SEARCH FROM KEY SEARCH
		DECLARE @v_index INT = 1;
		DECLARE @item NVARCHAR(max) = '';
		OPEN v_cur_condition;
		FETCH NEXT FROM v_cur_condition INTO @item
		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @v_index = 1
                SET @v_group_name = @item;
            ELSE IF @v_index = 2 BEGIN
                SET @v_group_type = @item;
			END;
			SET @v_index = @v_index + 1;
			FETCH NEXT FROM v_cur_condition INTO @item
		END;
		CLOSE v_cur_condition      
		DEALLOCATE v_cur_condition 
		
		--SET QUERY WITH KEY SEARCH
		IF LEN(@v_group_name) > 0 BEGIN
            SET @v_condition = @v_condition + ' AND UPPER(a.Group_Name) LIKE ''%'' + UPPER(cd.v_group_name) + ''%'' ';
        END;

		IF @v_group_type != '-1' AND LEN(@v_group_type) > 0 BEGIN
            SET @v_condition = @v_condition + ' AND a.typeMemberId = CD.v_typeMemberId';
        END;

        SET @v_sqlTableParamSearch = '
			WITH TableCondition AS 
			(
				SELECT @v_group_name AS v_group_name, @v_group_type AS v_group_type
			)
			';

		SET @v_sql = '
			SELECT T.* FROM
			(
				SELECT a.*, al1.Content Group_Type_Text, al2.Content Status_Text FROM CSE_Groups a

				LEFT JOIN CSE_Allcode al1 ON al1.CDName = ''GROUP'' AND al1.CDType = ''TYPE'' AND al1.CDVal = a.Group_Type
				LEFT JOIN CSE_Allcode al2 ON al1.CDName = ''GROUP'' AND al2.CDType = ''STATUS'' AND al2.CDVal = a.Status

				CROSS JOIN TableCondition CD
				
				WHERE a.Deleted = 0 '+ @v_condition + ' 
			)T ';

		--get total row by condition
		SET @v_sqlTotal_record = @v_sqlTableParamSearch + 'SELECT @p_total_record = COUNT(*) FROM (' + @v_sql + ') AS o';
		EXECUTE sp_executesql 
			@v_sqlTotal_record, 
			N'@p_total_record decimal(18,0) OUT, @v_group_name NVARCHAR(max), @v_group_type NVARCHAR(max)', 
			@p_total_record OUT, @v_group_name = @v_group_name, @v_group_type = @v_group_type;
		
		--paging list
		SET @v_sql = 'SELECT ROW_NUMBER() over(' + @v_orderby + ') STT, V.* FROM( ' + @v_sql + ') V ';

		IF @p_startrow > 0 AND @p_endrow > 0 
			--get by paging
            SET @v_sql = @v_sqlTableParamSearch + 'SELECT * FROM (' + @v_sql  + ') AS R WHERE STT BETWEEN @p_startrow AND @p_endrow ';	
        ELSE BEGIN
			--get all
            SET @v_sql = @v_sqlTableParamSearch + 'SELECT * FROM (' + @v_sql  + ') AS R WHERE @p_startrow = @p_startrow ';
		END;

		PRINT @v_sql;

		EXECUTE sp_executesql 
			@v_sql, 
			N'@v_group_name NVARCHAR(max), @v_group_type NVARCHAR(max), 
			@p_startrow INT, @p_endrow INT', 
			@v_group_name = @v_group_name, @v_group_type = @v_group_type, 
			@p_startrow = @p_startrow, @p_endrow = @p_endrow;
END
GO

CREATE PROCEDURE [dbo].[sp_user_search]
(
		@p_key_search NVARCHAR(max),
        @p_startrow INT,
        @p_endrow INT,
        @p_orderby VARCHAR(200),
        @p_total_record DECIMAL(18,0) OUT
)
AS
BEGIN
		DECLARE v_cur_condition CURSOR FOR 
		SELECT [str] AS condition FROM [dbo].[FN_ParserStrToRow](@p_key_search,'|');
		
        DECLARE @v_condition NVARCHAR(max) = '';
		DECLARE @v_sql NVARCHAR(max);
        DECLARE @v_sqlTableParamSearch NVARCHAR(max);
        DECLARE @v_sqlTotal_record NVARCHAR(max);
        DECLARE @v_orderby NVARCHAR(max) = 'ORDER BY ';

        DECLARE @v_user_name NVARCHAR(max) = '';
        DECLARE @v_user_type NVARCHAR(max) = '';
		DECLARE @v_status NVARCHAR(max) = '';	

		--SET ORDER BY
		IF LEN(@p_orderby) > 0 
           SET @v_orderby = @v_orderby + @p_orderby;
        ELSE BEGIN
           SET @v_orderby = @v_orderby + ' User_Id DESC';
        END;

		--GET AND SET INFO SEARCH FROM KEY SEARCH
		DECLARE @v_index INT = 1;
		DECLARE @item NVARCHAR(max) = '';
		OPEN v_cur_condition;
		FETCH NEXT FROM v_cur_condition INTO @item
		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @v_index = 1
                SET @v_user_name = @item;
			 ELSE IF @v_index = 2
                SET @v_user_type = @item;
            ELSE IF @v_index = 3 BEGIN
                SET @v_status = @item;
			END;
			SET @v_index = @v_index + 1;
			FETCH NEXT FROM v_cur_condition INTO @item
		END;
		CLOSE v_cur_condition      
		DEALLOCATE v_cur_condition 
		
		--SET QUERY WITH KEY SEARCH
		IF LEN(@v_user_name) > 0 BEGIN
            SET @v_condition = @v_condition + ' AND UPPER(a.User_Name) LIKE ''%'' + UPPER(cd.v_user_name) + ''%'' ';
        END;

		IF @v_user_type != '-1' AND LEN(@v_user_type) > 0 BEGIN
            SET @v_condition = @v_condition + ' AND a.User_Type = CD.v_user_type';
        END;

		IF @v_status != '-1' AND LEN(@v_status) > 0 BEGIN
            SET @v_condition = @v_condition + ' AND a.Status = CD.v_status';
        END;

        SET @v_sqlTableParamSearch = '
			WITH TableCondition AS 
			(
				SELECT @v_user_name AS v_user_name, @v_user_type AS v_user_type, @v_status AS v_status
			)
			';

		SET @v_sql = '
			SELECT T.* FROM
			(
				SELECT a.*, al1.Content User_Type_Text, al2.Content Status_Text FROM CSE_Users a

				LEFT JOIN CSE_Allcode al1 ON al1.CDName = ''GROUP'' AND al1.CDType = ''TYPE'' AND al1.CDVal = a.User_Type
				LEFT JOIN CSE_Allcode al2 ON al1.CDName = ''USER'' AND al2.CDType = ''STATUS'' AND al2.CDVal = a.Status

				CROSS JOIN TableCondition CD
				
				WHERE a.Deleted = 0 '+ @v_condition + ' 
			)T ';

		--get total row by condition
		SET @v_sqlTotal_record = @v_sqlTableParamSearch + 'SELECT @p_total_record = COUNT(*) FROM (' + @v_sql + ') AS o';
		EXECUTE sp_executesql 
			@v_sqlTotal_record, 
			N'@p_total_record decimal(18,0) OUT, @v_user_name NVARCHAR(max), @v_user_type NVARCHAR(max), @v_status NVARCHAR(max)', 
			@p_total_record OUT, @v_user_name = @v_user_name, @v_user_type = @v_user_type, @v_status = @v_status;
		
		--paging list
		SET @v_sql = 'SELECT ROW_NUMBER() over(' + @v_orderby + ') STT, V.* FROM( ' + @v_sql + ') V ';

		IF @p_startrow > 0 AND @p_endrow > 0 
			--get by paging
            SET @v_sql = @v_sqlTableParamSearch + 'SELECT * FROM (' + @v_sql  + ') AS R WHERE STT BETWEEN @p_startrow AND @p_endrow ';	
        ELSE BEGIN
			--get all
            SET @v_sql = @v_sqlTableParamSearch + 'SELECT * FROM (' + @v_sql  + ') AS R WHERE @p_startrow = @p_startrow ';
		END;

		PRINT @v_sql;

		EXECUTE sp_executesql 
			@v_sql, 
			N'@v_user_name NVARCHAR(max), @v_user_type NVARCHAR(max), @v_status NVARCHAR(max), 
			@p_startrow INT, @p_endrow INT', 
			@v_user_name = @v_user_name, @v_user_type = @v_user_type, @v_status = @v_status,
			@p_startrow = @p_startrow, @p_endrow = @p_endrow;
END
